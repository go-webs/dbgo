
func (dbGame Database) OrHavingBetween(col string, haystack ...any) Database {
	dbGame.OrHavingRaw(fmt.Sprintf("%s BETWEEN ? AND ?", util.BackQuotes(col)), haystack)
	return dbGame
}
func (dbGame Database) OrHavingNotBetween(col string, haystack ...any) Database {
	dbGame.OrHavingRaw(fmt.Sprintf("%s NOT BETWEEN ? AND ?", util.BackQuotes(col)), haystack)
	return dbGame
}
func (dbGame Database) OrHavingNull(col string) Database {
	dbGame.OrHavingRaw(fmt.Sprintf("%s IS NULL", util.BackQuotes(col)))
	return dbGame
}
func (dbGame Database) OrHavingNotNull(col string) Database {
	dbGame.OrHavingRaw(fmt.Sprintf("%s IS NOT NULL", util.BackQuotes(col)))
	return dbGame
}
